    <script>
        // DOM Ready
        document.addEventListener("DOMContentLoaded", function() {

            // 1. URL Parameter Personalization
            const params = new URLSearchParams(window.location.search);
            const city = params.get('city') || 'Your Area';
            const company = params.get('company') || 'Your Business';

            // Update all dynamic elements with smooth transition
            const updateDynamic = (selector, value) => {
                document.querySelectorAll(selector).forEach(el => {
                    el.style.transition = 'opacity 0.4s ease';
                    el.style.opacity = '0';
                    setTimeout(() => {
                        el.textContent = value;
                        el.style.opacity = '1';
                    }, 200);
                });
            };

            updateDynamic('.dynamic-city', city);
            updateDynamic('.dynamic-company', company);

            // 2. Scroll Reveal Animation (IntersectionObserver - performant)
            const revealElements = document.querySelectorAll('.reveal, .reveal-stagger');

            const revealObserver = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('active');
                        revealObserver.unobserve(entry.target);
                    }
                });
            }, {
                threshold: 0.1,
                rootMargin: '0px 0px -50px 0px'
            });

            revealElements.forEach(el => revealObserver.observe(el));

            // 3. Smooth scroll for anchor links
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const target = document.querySelector(this.getAttribute('href'));
                    if (target) {
                        target.scrollIntoView({ behavior: 'smooth', block: 'start' });
                    }
                });
            });
        });
    </script>
